    <header class="d-flex flex-wrap align-items-center gap-2 mb-4 no-print">
      <h1 class="h4 m-0"> Miesiczna Obecno Klienta</h1>
      <div class="ms-auto d-flex align-items-center gap-2">
        <select id="clientSelector" class="form-select" style="width: 250px;"></select>
        <input type="month" id="monthPicker" class="form-control" style="width: 180px;">
        <button id="printBtn" class="btn btn-outline-secondary">Drukuj</button>
      </div>

        <!-- Raport bdzie renderowany tutaj -->
    </div>
  </div>

<script>
document.addEventListener('DOMContentLoaded', () => {
    const API_URL = "/api";
    const clientSelector = document.getElementById('clientSelector');
    const monthPicker = document.getElementById('monthPicker');
    const reportContainer = document.getElementById('reportContainer');
    const printBtn = document.getElementById('printBtn');

    const showAlert = (msg, type = "success") => {
        const box = document.getElementById("alertBox");
        box.innerHTML = `<div class="alert alert-${type} alert-dismissible fade show" role="alert">
            ${msg}<button type="button" class="btn-close" data-bs-dismiss="alert"></button></div>`;
    };

    async function initialize() {
        const today = new Date();
        monthPicker.value = `${today.getFullYear()}-${String(today.getMonth() + 1).padStart(2, '0')}`;

        try {
            const response = await fetch(`${API_URL}/clients`);
            if (!response.ok) throw new Error('Bd adowania listy klient贸w.');
            const clients = await response.json();
            
            clientSelector.innerHTML = '<option value="">-- Wybierz klienta --</option>';
            clients
                .filter(c => c.active)
                .forEach(client => {
                    const option = new Option(client.full_name, client.client_id);
                    clientSelector.add(option);
                });
        } catch (error) {
            showAlert(error.message, 'danger');
        }

        clientSelector.addEventListener('change', loadReport);
        monthPicker.addEventListener('change', loadReport);
        printBtn.addEventListener('click', () => window.print());
    }

    async function loadReport() {
        const clientId = clientSelector.value;
        const month = monthPicker.value;

        if (!clientId || !month) {
            reportContainer.innerHTML = '<p class="text-muted text-center p-5">Wybierz klienta i miesic, aby zobaczy raport.</p>';
            return;
        }

        reportContainer.innerHTML = '<p class="text-muted text-center p-5">adowanie raportu...</p>';

        try {
            const response = await fetch(`${API_URL}/clients/${clientId}/attendance?month=${month}`);
            if (!response.ok) throw new Error(`Bd serwera: ${response.statusText}`);
            const data = await response.json();
            renderReport(data);
        } catch (error) {
            reportContainer.innerHTML = `<div class="alert alert-danger">${error.message}</div>`;
        }
    }

    function renderReport(data) {
        const clientName = clientSelector.options[clientSelector.selectedIndex].text;
        const monthName = new Date(monthPicker.value + '-02').toLocaleString('pl-PL', { month: 'long', year: 'numeric' });

        if (data.length === 0) {
            reportContainer.innerHTML = `
                <h3 class="mb-3">Obecno - ${clientName}</h3>
                <p><strong>Miesic:</strong> ${monthName}</p>
                <p class="text-muted">Brak zarejestrowanych obecnoci dla tego klienta w wybranym miesicu.</p>`;
            return;
        }
        
        const statusCounts = { obecny: 0, nieobecny: 0, sp贸藕niony: 0, usprawiedliwiony: 0 };
        const statusTranslations = {
            obecny: 'Obecny',
            nieobecny: 'Nieobecny',
            sp贸藕niony: 'Sp贸藕niony',
            usprawiedliwiony: 'Usprawiedliwiony'
        };

        const tableRows = data.map(item => {
            if (statusCounts.hasOwnProperty(item.status)) {
                statusCounts[item.status]++;
            }
            return `
                <tr>
                    <td>${new Date(item.session_date).toLocaleDateString('pl-PL')}</td>
                    <td>${item.group_name}</td>
                    <td>${item.topic_title}</td>
                    <td>${statusTranslations[item.status] || item.status}</td>
                </tr>
            `;
        }).join('');

        reportContainer.innerHTML = `
            <div class="d-flex justify-content-between align-items-start mb-4">
                <div>
                    <h3 class="mb-1">Raport Obecnoci</h3>
                    <div><strong>Klient:</strong> ${clientName}</div>
                    <div><strong>Miesic:</strong> ${monthName}</div>
                </div>
                <div class="text-end">
                    <small class="text-muted">Wygenerowano: ${new Date().toLocaleString("pl-PL")}</small>
                </div>
            </div>
            
            <table class="table table-bordered">
                <thead class="table-light">
                    <tr>
                        <th>Data</th>
                        <th>Grupa</th>
                        <th>Temat Sesji</th>
                        <th>Status</th>
                    </tr>
                </thead>
                <tbody>${tableRows}</tbody>
            </table>

            <div class="mt-4">
                <h4>Podsumowanie</h4>
                <p>
                    <span class="badge text-bg-success me-2">Obecny: ${statusCounts.obecny}</span>
                    <span class="badge text-bg-danger me-2">Nieobecny: ${statusCounts.nieobecny}</span>
                    <span class="badge text-bg-warning me-2">Sp贸藕niony: ${statusCounts.sp贸藕niony}</span>
                    <span class="badge text-bg-info me-2">Usprawiedliwiony: ${statusCounts.usprawiedliwiony}</span>
                </p>
            </div>
        `;
    }

    initialize();
});
</script>
